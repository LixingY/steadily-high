### 1. `el-input` 自定义规则检验时，错误提示信息为 `Error`，而不是自定义的

```js
// 整数 验证
const validateInt = (rule, value, callback) => {
  const intReg = /^-?[0-9]\d*$/;
  if (intReg.test(value)) {
    callback();
  } else {
    // 修改前 =>
    // callback(new Error("请输入整数"))

    // 修改后 =>
    callback("请输入整数");
  }
};
```
